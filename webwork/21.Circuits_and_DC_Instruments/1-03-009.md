from random import randrange
R1 = randrange(20,200,5)
R2 = randrange(45,195,5)
R3 = randrange(26,226,5)
R4 = randrange(32,232,5)
R = 1/(1/(R2) + 1/(R1 + R3)) + R4
emf = randrange(50,100,1)
current = emf/(R)
I4 = current
I2 = abs((emf - (I4)*(R4))/(R2))
I1 = abs(I4 - I2)
