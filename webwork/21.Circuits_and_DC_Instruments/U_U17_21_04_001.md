## Answer Section

from random import randrange
emfSI = randrange(28,32,1)
resistance = sprintf("%0.2f",randrange(0.90,1.1,0.01))
resistanceSI = resistance*10**6

currentSI = (emfSI)/(resistanceSI)
currentmuA = currentSI*10**6
