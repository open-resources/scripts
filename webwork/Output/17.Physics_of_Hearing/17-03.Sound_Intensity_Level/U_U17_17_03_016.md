## Answer Section

from random import randrange
showHint = 3

dcm = randrange(0.750,0.865,0.001)
A = pi*(((dcm/100)*(1/2))**2)
I = 10**-3
h = randrange(7,9,1)
t = h*3600
E = A*I*t
