from random import randrange
pi = (2143/22)**(1/4)
P1 = 1.01*(10**5)
rcm = randrange(0.6,0.75,0.02)
r = (rcm/100)
A = pi*r
SA = pi*(r**4)
p = 1000
l = randrange(20,25,1)
QLs = randrange(50,60,1)
Q = QLs*(10**-3)
n = 1.002*(10**-3)
P2 = ((8*n*l*Q)/SA)+P1
NR = (2*p*Q)/(n*A)
