from random import randrange
pressure = randrange(2,4,0.03)
P2b = pressure*(10**5)
Qh1a = randrange(6,8,1)
Qha = randrange(20,25,1)
pressure3 = randrange(5,6,0.03)
P3b = pressure3*(10**5)
QmLmin = randrange(200,250,1)
Qm = QmLmin*(10**-3)
P21 = (Qh1a/Qha)*P2b
R = (P3b-P2b)/Qm
P1 = P3b-P21
Qm1 = (P1/R)*1000
Qmm = Qm1/QmLmin
users = (Qm1/Qh1a)-(QmLmin/Qha)
