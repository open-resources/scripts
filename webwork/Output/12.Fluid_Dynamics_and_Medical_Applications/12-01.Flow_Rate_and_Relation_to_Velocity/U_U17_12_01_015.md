from random import randrange
da = randrange(1.20,2.15,0.03)
ra = (da*(1/2)*(1/100))**2
A = pi*ra
vo = randrange(0.4,0.6,0.011)
Qm = A*vo
Q = Qm*1000000
g = 9.80
x = randrange(0.134,0.264,0.001)
v2 = ((vo**2)+(2*g*x))**(1/2)
r = (Qm/(pi*v2))**(1/2)
d = 2*r*100
