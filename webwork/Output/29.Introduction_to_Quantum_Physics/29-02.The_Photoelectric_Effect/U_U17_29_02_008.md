## Answer Section

from random import randrange
showHint = 3

c = 3*10**8
e = 1.6*10**-19
h = 6.63*10**-34

energy = sprintf("%0.3f", randrange(0.5,0.75,0.025))
energySI = (energy*e)
wavelength = randrange(275,325,5)
wavelengthSI = wavelength*10**-9

phiSI = (h*c)/(wavelengthSI) - energySI
phi = (phiSI/e)
