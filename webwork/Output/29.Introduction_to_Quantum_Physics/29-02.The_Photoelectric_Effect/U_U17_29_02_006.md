## Answer Section

from random import randrange
showHint = 3

c = 3*10**8
e = 1.6*10**-19
h = 6.63*10**-34

phi = sprintf("%0.2f", randrange(4.75,5,0.05))
phiSI = (phi*e)
wavelength = randrange(100,150,5)
wavelengthSI = wavelength*10**-9

energySI = (h*c)/(wavelengthSI) - phiSI
energy = (energySI/e)
