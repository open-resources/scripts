## Answer Section

from random import randrange
showHint = 3

c = 3*10**8
e = 1.6*10**-19
h = 6.63*10**-34

frequency = randrange(1500,1600,10)
frequencySI = frequency*10**3

energySI = (h*frequencySI)
energy = (energySI/e)*10**9
