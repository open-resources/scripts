from random import randrange
showHint = 3

c = 3*10**8
e = 1.6*10**-19
h = 6.63*10**-34

energy = sprintf("%0.2f", randrange(1,1.5,0.05))
energySI = (energy*e)

frequencySI = (energySI)/(h)

wavelengthSI = (h*c)/(energySI)
wavelength = wavelengthSI*10**6
