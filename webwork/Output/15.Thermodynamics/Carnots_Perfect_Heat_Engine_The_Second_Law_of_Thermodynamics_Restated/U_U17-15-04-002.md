from random import randrange
temp = randrange(400,800,0.1)
temp2 = randrange(15,50,0.1)
eff = sprintf("%0.1f",randrange(40,50,0.1))

A1 = (((temp+273.15)-(temp2+273.15))/(temp+273.15))*100
A2 = (A1)/(eff)
