temp = randrange(-40,-20,0.1)
temp2 = randrange(30,60,0.1)
heat = randrange(4000,5000,0.1)
A1 = (temp+273.15)/((temp2+273.15)-(temp+273.15))
A2 = (heat/4.186)/(A1)
A3 = (A2)*(4186)*(10/(3600000))
