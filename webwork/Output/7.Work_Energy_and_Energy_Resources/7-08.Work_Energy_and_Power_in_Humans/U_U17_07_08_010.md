from random import randrange
showHints = 3

mj = randrange(75,99,3)
ml = randrange(12,19,1)
v = randrange(6,9,1)
v2 = v**2
dcm = randrange(1.30,3.75,0.05)
d = dcm/100
g = 9.8
Fleg = (ml*v2)/(2*d)
Fj = mj*g
Fnet = Fleg+Fj
Force = Fnet/Fj
