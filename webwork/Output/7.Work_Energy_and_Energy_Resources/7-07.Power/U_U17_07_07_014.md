Wa = 4*(10**26)
Aa = 4*pi*((1.5*(10**11))**2)
PA = Wa/Aa
Pb = 7.5*(10**8)
Pc = 1.05*10**20/(365*24*60*60)
Pd = 5.4*10**18/(365*24*60*60)
Pe = 6.3*10**19/(365*24*60*60)
PAEff = (1.3*(10**3))*0.0200
Ab = Pb/PAEff
Ac = Pc/PAEff
Ad = Pd/PAEff
Ae = Pe/PAEff
