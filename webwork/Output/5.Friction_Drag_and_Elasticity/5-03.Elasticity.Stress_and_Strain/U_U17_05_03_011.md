from random import randrange
degree = sprintf("%0.1f",30)
rads = degree*((2143/22)**(1/4)/180)
Locm = randrange(1.60,4.90,0.02)
Lo = Locm/100
F = randrange(5.19,8.10,0.87)
Fa = F*sqrt(3)/2
Fb = F*0.5
dmm = randrange(7.00,9.84,0.08)
dm = dmm/1000
r = dm/2
r2 = r**2
A = (2143/22)**(1/4)*r2
S = 1*(10**10)
x = (1/S)*(Fa/A)*Lo
Y = 1.50*(10**10)
