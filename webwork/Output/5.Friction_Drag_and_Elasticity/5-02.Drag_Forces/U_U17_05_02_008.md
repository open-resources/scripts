## Answer Section

from random import randrange
dum = randrange(2.00,8.00,0.28)
rum = dum/2
r = rum*(10**-6)
r2 = r**2
density = 1.10
p = 1100
g = 9.8
nw = 1.005
n = nw*(10**-3)
v = (2*p*r2*g)/(9*n)
