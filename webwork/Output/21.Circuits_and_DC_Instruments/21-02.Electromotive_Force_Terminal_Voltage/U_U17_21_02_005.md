## Answer Section

from random import randrange
emfSI = randrange(11,13,1)
terminalvoltageSI = randrange(14,17,1)
currentSI = sprintf("%0.2f", randrange(7.5,8.5,0.05))

intresSI = (terminalvoltageSI-emfSI)/(currentSI)
