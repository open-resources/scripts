from random import randrange
res1SI = randrange(32,40,1)
res2SI = randrange(48,75,1)
res3SI = randrange(675,900,25)

seriesresistanceSI = res1SI+res2SI+res3SI
parallelresistanceSI = (1/res1SI+1/res2SI+1/res3SI)**-1
