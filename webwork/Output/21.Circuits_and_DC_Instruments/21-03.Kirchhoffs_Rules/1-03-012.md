from random import randrange
R1 = randrange(35,210,5)
R2 = randrange(55,205,5)
R3 = randrange(27,222,5)
R4 = randrange(39,234,5)
R = R1 + 1/(1/(R4) + 1/(R2 + R3))
emf = randrange(80,140,1)
current = emf/(R)
I1 = current
I4 = abs((emf - (I1)*(R1))/(R4))
I2 = abs(I1 - I4)
I3 = I2
