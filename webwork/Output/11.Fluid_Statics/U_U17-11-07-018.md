from random import randrange
mass = randrange(60,80,0.1)
percent = randrange(2,4,0.01)
percent2 = randrange(4.01,7,0.01)
a = (1-((percent)/100))
b = (1-((percent2)/100))

A1 = (mass)*((1/(b))-(1/a))
