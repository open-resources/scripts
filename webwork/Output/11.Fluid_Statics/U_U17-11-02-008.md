## Answer Section

from random import randrange
mass = randrange(2,3,0.01)
volume = randrange(17,23,0.1)


m = (mass)+(volume)*(680)*(0.001)
v = (volume)*(0.001)+(mass)/(7800)
A1 = (m)/(v)
