from random import randrange
showHint = 3

dcm = randrange(8,9,0.01)
r = ((dcm/2)*(1/100))**2
dmm = randrange(2,3,1)
d = dmm*(10**-3)
P = randrange(50,60,1)
A = pi*r
k = 0.84
T2 = (P*d)/(k*A)
T1 = randrange(50,64,1)
T = T2+T1
t = 60
Lv = 2340
m = (P*t)/Lv
