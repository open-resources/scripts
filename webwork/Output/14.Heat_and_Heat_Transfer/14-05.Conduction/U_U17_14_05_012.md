## Answer Section

showHint = 3

dcm = 15
d = dcm/100
l = 10
w = 15
h = 3
Awallsandfloor = (l*h*2)+(w*h*2)+(l*w)
Aattic = (l*w)
k = 0.042
Qt1 = (k/d)*(Awallsandfloor+Aattic)
Qt2 = (k/d)*(Awallsandfloor)+(k/(d+0.08))*(Aattic)
percent = abs(((Qt2-Qt1)/Qt1)*100)
