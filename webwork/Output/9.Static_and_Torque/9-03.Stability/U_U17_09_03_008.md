## Answer Section

mcar = 900
mbridge = 2500
g = 9.8
alpha = 40
length = 9
dcombridge = 1.5
Ttotal = ((g)/(length*sin(alpha*pi/180)))*((mbridge)*(dcombridge)+0.5*(mcar)*(length))
Tpercable = Ttotal/2
FV = (g)*(mbridge+mcar)-(Ttotal)*sin(alpha*pi/180)
FH = (Ttotal)*cos(alpha*pi/180)
gamma = atan(FV/FH)*(180/pi)
