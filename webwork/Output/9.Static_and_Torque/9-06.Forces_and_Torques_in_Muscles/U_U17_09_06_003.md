## Answer Section

alpha = 180-55
beta = 180+75
FQ = 1250
Fx = FQ*cos(alpha*(pi/180))+FQ*cos(beta*(pi/180))
Fy = FQ*sin(alpha*(pi/180))+FQ*sin(beta*(pi/180))
F = ((Fx**2)+(Fy**2))**(1/2)
Fyp = -Fy
theta = arctan((Fy/Fx))*(180/pi)+180
