from random import randrange
showHint = 3

X = randrange (15,25,1)
f = randrange (0.4,2.0,0.1)
m = randrange (0.5,1.5,0.1)
k = m*(2*pi*f)**2
t1 = 1.0
t2 = 2.0
t3 = 3.0
x1 = X*cos(2*pi*f*t1)
x2 = X*cos(2*pi*f*t2)
x3 = X*cos(2*pi*f*t3)
v1 = (-2*pi*f*X)*sin(2*pi*f*t1)
v2 = (-2*pi*f*X)*sin(2*pi*f*t2)
v3 = (-2*pi*f*X)*sin(2*pi*f*t3)
K1 = 0.5*m*(v1/100)**2
K2 = 0.5*m*(v2/100)**2
K3 = 0.5*m*(v3/100)**2
U1 = 0.5*k*(x1/100)**2
U2 = 0.5*k*(x2/100)**2
U3 = 0.5*k*(x3/100)**2
E1 = K1 + U1
E2 = K2 + U2
E3 = K3 + U3
