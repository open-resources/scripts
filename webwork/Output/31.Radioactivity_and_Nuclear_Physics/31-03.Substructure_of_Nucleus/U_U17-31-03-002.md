from random import randrange
showHint = 3

p_nuclearSI = 2.3*10**17

massSI = sprintf("%0.2f", randrange(1,1.5,0.05))

volumeSI = (massSI/p_nuclearSI)
lengthSI = (volumeSI)**(1/3)
length = lengthSI*10**6
