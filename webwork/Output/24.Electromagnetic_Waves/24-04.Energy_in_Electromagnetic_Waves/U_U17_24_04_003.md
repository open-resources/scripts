from random import randrange
output = randrange(0.2,0.3,0.01)
diameter = randrange(0.95,1.05,0.01)
radius = (diameter*10**-3)/2

E = (output*10**-3)/(pi*radius**2)
F = ((2*(4*pi*10**-7)*E)/(3*10**8))**(1/2)
G = 3*10**8*F
