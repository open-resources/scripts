speed = randrange(5.95,6.05,0.01)

speed_proton = (((2*(1.6*10**-19)*(E*10**-24))/(1.67*10**-27))**(1/2))*10**8
F = (((1.67*10**-27)*(speed_proton*10**-8))/((1.6*10**-19)*strength))*10**16
