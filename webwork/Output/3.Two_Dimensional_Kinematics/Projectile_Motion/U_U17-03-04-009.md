from random import randrange
h = randrange(50,70,2)
d = randrange(90,110,4)
g = 9.8

t = sqrt(2*h/g)
b = d/t
c = g*t
d1 = sqrt(b**2+c**2)
d2 = arctan(c/b)*180/pi
