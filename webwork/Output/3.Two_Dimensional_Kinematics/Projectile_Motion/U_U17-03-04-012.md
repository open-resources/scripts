from random import randrange
d = randrange(5,10,1)
v = randrange(10,15,1)
g = 9.8

A1 = 0.5*180/pi*arcsin(g*d/v**2)
A2 = 90-A1
vx = v*cos(A1*pi/180)
A3 = d/vx
