from random import randrange
va = randrange(275,285,1)
d = randrange(2995,3005,1)
a = randrange(4.5,5.5,0.1)
A1 = d*1000/(1.5*3600)

x = A1*cos(a*pi/180)-va
y= -A1*sin(a*pi/180)
p1 = (x)**2
p2 = (y)**2

A2 = sqrt(p1+p2)
A3 = atan(-y/x)*180/pi
