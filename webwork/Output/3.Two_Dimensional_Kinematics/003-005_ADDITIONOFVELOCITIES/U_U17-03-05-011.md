from random import randrange
s = randrange(2.10,2.30,0.01)
b = randrange(27.0,33.0,1)
v = randrange(4.25,4.75,0.01)
a = randrange(47.0,53.0,1)
c = 90-b

x = -v*cos(a*pi/180)-s*cos(c*pi/180)
y = -v*sin(a*pi/180)-s*sin(c*pi/180)
p1 = (x)**2
p2 = (y)**2

A1 = sqrt(p1+p2)
A2 = atan(y/x)*180/pi
