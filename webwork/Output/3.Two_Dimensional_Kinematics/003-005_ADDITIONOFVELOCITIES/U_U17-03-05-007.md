from random import randrange
a= randrange(183,186,1)
sw = a-180
as = randrange(255,265,1)
js = randrange(33,37,1)
b = randrange(13,17,1)
rad1 = a*pi/180
rad2 = b*pi/180
x = as*cos(rad1)+js*cos(rad2)
y = as*sin(rad1)+js*sin(-rad2)
p1 = (x)**2
p2 = (y)**2
z = sqrt(p1+p2)
t = atan(y/x)*180/pi
