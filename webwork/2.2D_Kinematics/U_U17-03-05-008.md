## Answer Section

from random import randrange
vw = randrange(1.3,1.7,0.1)
degw = randrange(37,43,1)
vb = randrange(6.7,7.3,0.1)
rad = degw*pi/180

A1 = asin(vw/vb*cos(rad))*180/pi
A2 = vb*cos(A1*pi/180)+vw*sin(rad)
