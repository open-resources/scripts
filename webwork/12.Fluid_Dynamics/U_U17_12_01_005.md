## Answer Section

da = 20

db = 40
wb = 60
Aa = (da)*(wa)
Ab = (db)*(wb)
QLs1 = sprintf("%0.1f",3)
Q = (QLs1*10**5)*(1/1000)

va = Q/Aa
vb = Q/Ab
