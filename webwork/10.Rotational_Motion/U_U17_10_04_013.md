from random import randrange
showHint = 3

heighto = randrange(32,36,1)
velocity = randrange(30,38,1)
mass = randrange(200,280,1)
gravity = 9.8
Mass = 100
Radius = 0.330
Inertia = 1.1238
heightf = ((velocity**2)/(mass*gravity))*(mass/2+(Inertia/(Radius**2)))
Energy = mass*gravity*(heightf-heighto)
