from random import randrange
power = randrange(48.5,52.5,0.1)
distance = randrange(25,35,0.1)

E = ((power/2)*10**3)/(2*pi*(distance*10**3)**2)
F = ((2*E)/((3*10**8)*(8.85*10**-12)))**(1/2)
