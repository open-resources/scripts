from random import randrange
capacitance = randrange(4.75,5.25,0.01)
wavelength = randrange(3.1,3.5,0.01)

E = (3*10**8)/wavelength
F = (1/(4*(pi**2)*(capacitance*10**-12)*E**2))*10**6
