## Answer Section

from random import randrange
showHint = 3

e = 0.8
o = 5.67*(10**-8)
T2C = randrange(20,28,1)
T2 = (T2C+273.15)**4
QtkW = randrange(20,30,1)
Qt = QtkW*(10**3)
dcm = randrange(1.02,1.23,0.01)
r = ((dcm/100)*(1/2))**2
A = 4*pi*r
T1 = (T2-((-Qt)/(o*e*A)))**(1/4)
