## Answer Section

from random import randrange
diameter = sprintf("%0.1f",randrange(9,20,0.1))
radiusm = diameter/200
charge = sprintf("%0.1f",randrange(7.5,8.5,0.1))
K = 9*10**9

voltage = ((K)*(charge))/(radiusm)
