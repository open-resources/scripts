## Answer Section

from random import randrange
showHint = 3

c = 3*10**8
e = 1.6*10**-19
h = 6.63*10**-34
hbar = h/(2*pi)

Utime = sprintf("%0.2f", randrange(2.75,3.25,0.05))
UtimeSI = Utime*10**-3

UenergySI = (hbar/2)/(UtimeSI)
Uenergy = (UenergySI/e)*10**13
