## Answer Section

from random import randrange
showHint = 3

e = 1.6*10**-19
h = 6.63*10**-34

energy = sprintf("%0.3f", randrange(0.315,0.345,0.005))
energySI = (energy*e)

frequencySI = (energySI)/(h)
frequency = frequencySI*10**-13
