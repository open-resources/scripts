from random import randrange
showHint = 3

c = 3*10**8
e = 1.6*10**-19
h = 6.63*10**-34
hbar = h/(2*pi)

massSI = 9.11*10**-31
Uposition = sprintf("%0.2f", randrange(1,1.5,0.05))
UpositionSI = Uposition*10**-6

UvelocitySI = (hbar/2)/(massSI*UpositionSI)
energySI = 0.5*(massSI*(UvelocitySI**2))
energy = (energySI/e)
