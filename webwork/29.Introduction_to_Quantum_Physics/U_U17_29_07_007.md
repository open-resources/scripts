## Answer Section

from random import randrange
showHint = 3

c = 3*10**8
e = 1.6*10**-19
h = 6.63*10**-34
hbar = h/(2*pi)

Uenergy = sprintf("%0.2f", randrange(1,1.5,0.05))
UenergySI = (Uenergy*e)*10**6

UtimeSI = (hbar/2)/(UenergySI)
Utime = UtimeSI*10**22
