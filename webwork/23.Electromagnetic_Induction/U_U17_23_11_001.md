## Answer Section

from random import randrange
inductance = sprintf("%0.1f", randrange(29.5,30.5,0.1))
inductanceSI = inductance*10**-3
reactanceSI = randrange(90,110,2)

frequencySI = reactanceSI/(2*pi*(inductanceSI))
