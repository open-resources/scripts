## Answer Section

from random import randrange
m = randrange(34.1,68.3,1.25)
b = randrange(1.30,4.60,0.33)
Q = be-6
d= m/63.5
Np = d*(6.02e23)*(29)
qe = 1.6e-19
Ne = Q/qe
f = Ne/Np
