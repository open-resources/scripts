from random import randrange
showHint = 3

Mass = randrange(1,2,0.1)
heightkm = randrange(10,20,1)
height = heightkm*1000
c = 2.998*(10**8)
gravity = 9.8
Energy = Mass*(c**2)
mass = Energy/(gravity*height)
