## Answer Section

from random import randrange
showHint = 3

q = 1.6*10**-19
mc = 938.3
MV = randrange(50,60,1)
V = MV*10**6
meV = 1/(1.6*10**-13)
gammaa = ((q*V*meV)+mc)/mc
velocitya = (1-(1/(gammaa**2)))**(1/2)
gammab = ((q*V*meV)+0.511)/0.511
velocityb = (1-(1/(gammab**2)))**(1/2)
