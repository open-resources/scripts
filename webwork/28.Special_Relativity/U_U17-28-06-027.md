## Answer Section

showHint = 3

SunEnergy = 4*(10**26)
c = 2.998*(10**8)
kgs = SunEnergy/(6.4*(10**14))
timeseconds = 0.450*((1.99*(10**30))/(kgs))
years = timeseconds/(3.1536*(10**7))
massc = SunEnergy/(c**2)
percentd = ((massc*timeseconds)/(1.99*(10**30)))*100
