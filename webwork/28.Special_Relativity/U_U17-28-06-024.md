## Answer Section

from random import randrange
showHint = 3

c = 2.998*(10**8)
percent = randrange(34,40,1)
mass = randrange(1,2,0.1)
kWh = randrange(500,600,1)
monthlyenergy = kWh*3600*1000
Energy = (percent/100)*(mass/1000)*(c**2)
years = (Energy/monthlyenergy)*(1/12)
