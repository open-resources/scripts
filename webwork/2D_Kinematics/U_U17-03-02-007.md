## Answer Section

from random import randrange
a = randrange(10,40,0.1)
b = randrange(10,40,0.1)

a1 = ((a)*(sin(20*pi/180)))-((b)*(cos(40*pi/180)))
a2 = ((a)*(cos(20*pi/180)))+((b)*(sin(40*pi/180)))
A1 = sqrt(((a1)**(2))+((a2)**(2)))
A2 = atan(a1/a2)*(180/pi)
